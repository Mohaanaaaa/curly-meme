secure-print-app/
├── app/
│   ├── main.py              # Entry point for the Flask app
│   ├── models.py            # Database logic (File metadata & codes)
│   ├── utils/
│   │   ├── converter.py     # LibreOffice conversion logic
│   │   └── sanitizer.py     # Security checks & file renaming
│   ├── static/              # CSS, JS, and Images
│   │   ├── css/             # Tailwind CSS or custom styles
│   │   └── js/              # Frontend logic (Drag & Drop)
│   └── templates/           # HTML Interfaces
│       ├── customer/        # upload.html, settings.html, ticket.html
│       └── shop/            # dashboard.html
├── uploads/                 # TEMPORARY storage for user files
├── tests/                   # Security and unit tests
├── .env                     # Secret keys and config variables
├── requirements.txt         # Python dependencies
└── config.py                # App configuration (Paths, allowed formats)
